001package org.xbib.elasticsearch.common.langdetect;
002
003import org.elasticsearch.common.io.Streams;
004import org.testng.Assert;
005import org.testng.annotations.Test;
006
007import java.io.IOException;
008import java.io.InputStreamReader;
009import java.io.Reader;
010import java.io.StringWriter;
011import java.io.Writer;
012
013public class DetectLanguageTest extends Assert {
014
015    @Test
016    public void testEnglish() throws IOException {
017        testLanguage("/english.txt", "en");
018    }
019
020    @Test
021    public void testChinese() throws IOException {
022        testLanguage("/chinese.txt", "zh-cn");
023    }
024
025    @Test
026    public void testJapanese() throws IOException {
027        testLanguage("/japanese.txt", "ja");
028    }
029
030    @Test
031    public void testKorean() throws IOException {
032        testLanguage("/korean.txt", "ko");
033    }
034
035    private void testLanguage(String path, String lang) throws IOException {
036        Reader reader = new InputStreamReader(getClass().getResourceAsStream(path));
037        Writer writer = new StringWriter();
038        Streams.copy(reader, writer);
039        reader.close();
040        writer.close();
041        Detector detect = new Detector();
042        detect.loadDefaultProfiles();
043        assertEquals(detect.detect(writer.toString()), lang);
044    }
045
046}
