001package org.xbib.elasticsearch.common.langdetect;
002
003import org.testng.Assert;
004import org.testng.annotations.Test;
005
006public class LangProfileTest extends Assert {
007
008    @Test
009    public final void testLangProfile() {
010        LangProfile profile = new LangProfile();
011        assertEquals(profile.name, null);
012    }
013
014    @Test
015    public final void testLangProfileStringInt() {
016        LangProfile profile = new LangProfile("en");
017        assertEquals(profile.name, "en");
018    }
019
020    @Test
021    public final void testAdd() {
022        LangProfile profile = new LangProfile("en");
023        profile.add("a");
024        assertEquals((int) profile.freq.get("a"), 1);
025        profile.add("a");
026        assertEquals((int) profile.freq.get("a"), 2);
027        profile.omitLessFreq();
028    }
029
030    @Test
031    public final void testAddIllegally1() {
032        LangProfile profile = new LangProfile();
033        profile.add("a");
034        assertEquals(profile.freq.get("a"), null);
035    }
036
037    @Test
038    public final void testAddIllegally2() {
039        LangProfile profile = new LangProfile("en");
040        profile.add("a");
041        profile.add("");
042        profile.add("abcd");
043        assertEquals((int) profile.freq.get("a"), 1);
044        assertEquals(profile.freq.get(""), null);
045        assertEquals(profile.freq.get("abcd"), null);
046
047    }
048
049    @Test
050    public final void testOmitLessFreq() {
051        LangProfile profile = new LangProfile("en");
052        String[] grams = "a b c \u3042 \u3044 \u3046 \u3048 \u304a \u304b \u304c \u304d \u304e \u304f".split(" ");
053        for (int i = 0; i < 5; ++i) {
054            for (String g : grams) {
055                profile.add(g);
056            }
057        }
058        profile.add("\u3050");
059
060        assertEquals((int) profile.freq.get("a"), 5);
061        assertEquals((int) profile.freq.get("\u3042"), 5);
062        assertEquals((int) profile.freq.get("\u3050"), 1);
063        profile.omitLessFreq();
064        assertEquals(profile.freq.get("a"), null);
065        assertEquals((int) profile.freq.get("\u3042"), 5);
066        assertEquals(profile.freq.get("\u3050"), null);
067    }
068
069    @Test
070    public final void testOmitLessFreqIllegally() {
071        LangProfile profile = new LangProfile();
072        profile.omitLessFreq();
073    }
074}
