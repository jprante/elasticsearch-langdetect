001package org.xbib.elasticsearch.common.langdetect;
002
003import org.testng.Assert;
004import org.testng.annotations.Test;
005
006public class LanguageTest extends Assert {
007
008    @Test
009    public final void testLanguage() {
010        Language lang = new Language(null, 0);
011        assertEquals(lang.getLanguage(), null);
012        assertEquals(lang.getProbability(), 0.0, 0.0001);
013        assertEquals(lang.getLanguage(), null);
014
015        Language lang2 = new Language("en", 1.0);
016        assertEquals(lang2.getLanguage(), "en");
017        assertEquals(lang2.getProbability(), 1.0, 0.0001);
018        assertEquals(lang2.getLanguage(), "en");
019        assertEquals(lang2.getProbability(), 1.0);        
020    }
021}
