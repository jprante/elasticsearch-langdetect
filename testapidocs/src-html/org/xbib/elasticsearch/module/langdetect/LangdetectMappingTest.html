029import static org.elasticsearch.common.io.Streams.copyToStringFromClasspath;
030import static org.elasticsearch.common.xcontent.XContentFactory.jsonBuilder;
031
032public class LangdetectMappingTest extends Assert {
033
034    private DocumentMapperParser mapperParser;
035
036    @BeforeClass
037    public void setupMapperParser() throws IOException {
038        Index index = new Index("test");
039        Map<String, AnalyzerProviderFactory> analyzerFactoryFactories = Maps.newHashMap();
040        analyzerFactoryFactories.put("keyword",
041                new PreBuiltAnalyzerProviderFactory("keyword", AnalyzerScope.INDEX, new KeywordAnalyzer()));
042        Settings settings = ImmutableSettings.Builder.EMPTY_SETTINGS;
043        AnalysisService analysisService = new AnalysisService(index, settings, null, analyzerFactoryFactories, null, null, null);
044        mapperParser = new DocumentMapperParser(index,
045                settings,
046                analysisService,
047                new PostingsFormatService(index),
048                new DocValuesFormatService(index),
049                new SimilarityLookupService(index, settings),
050                null);
051        Detector detector = new Detector(settings);
052        detector.start();
053        mapperParser.putTypeParser(LangdetectMapper.CONTENT_TYPE,
054                new LangdetectMapper.TypeParser(detector));
055    }
056
057    @Test
058    public void testSimpleMappings() throws Exception {
059        String mapping = copyToStringFromClasspath("/simple-mapping.json");
060        DocumentMapper docMapper = mapperParser.parse(mapping);
061
062        String sampleText = copyToStringFromClasspath("/english.txt");
063        BytesReference json = jsonBuilder().startObject().field("_id", 1).field("someField", sampleText).endObject().bytes();
064        ParseContext.Document doc = docMapper.parse(json).rootDoc();
065
066        assertEquals(doc.get(docMapper.mappers().smartName("someField").mapper().names().indexName()), sampleText);
067        assertEquals(doc.getFields("someField.lang").length, 1);
068        assertEquals(doc.getFields("someField.lang")[0].stringValue(), "en");
069
070        // re-parse it
071        String builtMapping = docMapper.mappingSource().string();
072        docMapper = mapperParser.parse(builtMapping);
073
074        json = jsonBuilder().startObject().field("_id", 1).field("someField", sampleText).endObject().bytes();
075        doc = docMapper.parse(json).rootDoc();
076
077        assertEquals(doc.get(docMapper.mappers().smartName("someField").mapper().names().indexName()), sampleText);
078        assertEquals(doc.getFields("someField.lang").length, 1);
079        assertEquals(doc.getFields("someField.lang")[0].stringValue(), "en");
080    }
081
