002package org.xbib.elasticsearch.index.mapper.langdetect;
003
004import org.apache.lucene.util.ToStringUtils;
005import org.elasticsearch.common.xcontent.XContentBuilder;
006import org.elasticsearch.common.xcontent.XContentParser;
007import org.elasticsearch.index.analysis.AnalysisService;
008import org.elasticsearch.index.mapper.*;
009import org.elasticsearch.index.mapper.core.StringFieldMapper;
010import org.xbib.elasticsearch.common.langdetect.Detector;
011import org.xbib.elasticsearch.common.langdetect.Language;
012import org.xbib.elasticsearch.common.langdetect.LanguageDetectionException;
013
014import java.io.IOException;
015import java.nio.charset.Charset;
016import java.util.List;
017import java.util.Map;
018
019import static org.elasticsearch.index.mapper.MapperBuilders.stringField;
020
021public class LangdetectMapper implements Mapper {
022
023    public static final String CONTENT_TYPE = "langdetect";
024
025    public static class Builder extends Mapper.Builder<Builder, LangdetectMapper> {
026
027        private StringFieldMapper.Builder contentBuilder;
028        private StringFieldMapper.Builder langBuilder = stringField("lang");
029        private Detector detector;
030
031        public Builder(String name, Detector detector) {
032            super(name);
033            this.detector = detector;
034            this.contentBuilder = stringField(name);
035            this.builder = this;
036        }
037
038        public Builder content(StringFieldMapper.Builder content) {
039            this.contentBuilder = content;
040            return this;
041        }
042
043        public Builder lang(StringFieldMapper.Builder lang) {
044            this.langBuilder = lang;
045            return this;
046        }
047
048        @Override
049        public LangdetectMapper build(BuilderContext context) {
050            context.path().add(name);
051            StringFieldMapper contentMapper = contentBuilder.build(context);
052            StringFieldMapper langMapper = langBuilder.build(context);
053            context.path().remove();
054            return new LangdetectMapper(name, detector, contentMapper, langMapper);
055        }
056    }
057
058    public static class TypeParser implements Mapper.TypeParser {
059
060        private Detector detector;
061
062        public TypeParser(Detector detector) {
063            this.detector = detector;
064        }
065
066        @SuppressWarnings({"unchecked","rawtypes"})
067        @Override
068        public Mapper.Builder parse(String name, Map<String, Object> node, ParserContext parserContext)
069                throws MapperParsingException {
070            LangdetectMapper.Builder builder = new Builder(name, detector);
071            for (Map.Entry<String, Object> entry : node.entrySet()) {
072                String fieldName = entry.getKey();
073                Object fieldNode = entry.getValue();
074
075                if (fieldName.equals("fields")) {
076                    Map<String, Object> fieldsNode = (Map<String, Object>) fieldNode;
077                    for (Map.Entry<String, Object> fieldsEntry : fieldsNode.entrySet()) {
078                        String propName = fieldsEntry.getKey();
079                        Object propNode = fieldsEntry.getValue();
080
081                        if (name.equals(propName)) {
082                            builder.content((StringFieldMapper.Builder) parserContext.typeParser("string").parse(name,
083                                    (Map<String, Object>) propNode, parserContext));
084                        } else if ("lang".equals(propName)) {
085                            builder.lang((StringFieldMapper.Builder) parserContext.typeParser("string").parse("lang",
086                                    (Map<String, Object>) propNode, parserContext));
087                        }
088                    }
089                }
090            }
091
092            return builder;
093        }
094    }
095
096    private final String name;
097    private final Detector detector;
098    private final StringFieldMapper contentMapper;
099    private final StringFieldMapper langMapper;
100
101    public LangdetectMapper(String name, Detector detector, StringFieldMapper contentMapper, StringFieldMapper langMapper) {
102        this.name = name;
103        this.detector = detector;
104        this.contentMapper = contentMapper;
105        this.langMapper = langMapper;
106    }
107
108    @Override
109    public String name() {
110        return name;
111    }
112
113    @Override
114    public void parse(ParseContext context) throws IOException {
115        String content = null;
116
117        XContentParser parser = context.parser();
118        XContentParser.Token token = parser.currentToken();
119
120        if (token == XContentParser.Token.VALUE_STRING) {
121            // try decode UTF-8 base64 (e.g. from attachment mapper plugin)
122            content = parser.text();
123            try {
124                byte[] b = parser.binaryValue();
125                if (b != null && b.length > 0) {
126                    content = new String(b, Charset.forName("UTF-8"));
127                }
128            } catch (Exception e) {
129            }
130        }
131
132        context.externalValue(content);
133        contentMapper.parse(context);
134
135        if (content == null) {
136            return;
137        }
138
139        try {
140            List<Language> langs = detector.detectAll(content);
141            for (Language lang : langs) {
142                context.externalValue(lang.getLanguage());
143                langMapper.parse(context);
144            }
145        } catch(LanguageDetectionException e) {
146            context.externalValue("unknown");
147            langMapper.parse(context);
148        }
149    }
150
151    @Override
152    public void merge(Mapper mergeWith, MergeContext mergeContext) throws MergeMappingException {
153    }
154
155    @Override
156    public void traverse(FieldMapperListener fieldMapperListener) {
157        contentMapper.traverse(fieldMapperListener);
158        langMapper.traverse(fieldMapperListener);
159    }
160
161    @Override
162    public void traverse(ObjectMapperListener objectMapperListener) {
163    }
164
165    @Override
166    public void close() {
167        contentMapper.close();
168        langMapper.close();
169    }
170
171    @Override
172    public XContentBuilder toXContent(XContentBuilder builder, Params params) throws IOException {
173        builder.startObject(name);
174        builder.field("type", CONTENT_TYPE);
175
176        builder.startObject("fields");
177        contentMapper.toXContent(builder, params);
178        langMapper.toXContent(builder, params);
179        builder.endObject();
180
181        builder.endObject();
182        return builder;
183    }
184}
