001package org.xbib.elasticsearch.plugin.langdetect;
002
003import org.elasticsearch.common.inject.Inject;
004import org.elasticsearch.common.settings.Settings;
005import org.elasticsearch.index.AbstractIndexComponent;
006import org.elasticsearch.index.Index;
007import org.elasticsearch.index.mapper.MapperService;
008import org.elasticsearch.index.settings.IndexSettings;
009
010import org.xbib.elasticsearch.common.langdetect.Detector;
011import org.xbib.elasticsearch.index.mapper.langdetect.LangdetectMapper;
012
013public class RegisterLangdetectType extends AbstractIndexComponent {
014
015    @Inject
016    public RegisterLangdetectType(Index index, @IndexSettings Settings indexSettings,
017                                MapperService mapperService, Detector detector) {
018        super(index, indexSettings);
019        mapperService.documentMapperParser().putTypeParser("langdetect",
020                new LangdetectMapper.TypeParser(detector));
021    }
022}
