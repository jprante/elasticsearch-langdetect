002package org.xbib.elasticsearch.common.langdetect;
003
004import java.util.MissingResourceException;
005import java.util.ResourceBundle;
006
007public class Messages {
008
009    private static final ResourceBundle RESOURCE_BUNDLE =
010            ResourceBundle.getBundle(Messages.class.getPackage().getName() + ".messages");
011
012    private Messages() {
013    }
014
015    public static String getString(String key) {
016        try {
017            return RESOURCE_BUNDLE.getString(key);
018        } catch (MissingResourceException e) {
019            return '!' + key + '!';
020        }
021    }
022}
