020    public NGram() {
021        grams = new StringBuilder(" ");
022        capitalword = false;
023    }
024
025    public void addChar(char ch) {
026        ch = normalize(ch);
027        char lastchar = grams.charAt(grams.length() - 1);
028        if (lastchar == ' ') {
029            grams = new StringBuilder(" ");
030            capitalword = false;
031            if (ch == ' ') {
032                return;
033            }
034        } else if (grams.length() >= N_GRAM) {
035            grams.deleteCharAt(0);
036        }
037        grams.append(ch);
038        if (Character.isUpperCase(ch)) {
039            if (Character.isUpperCase(lastchar)) {
040                capitalword = true;
041            }
042        } else {
043            capitalword = false;
044        }
045    }
046
047    public String get(int n) {
048        if (capitalword) {
049            return null;
050        }
051        int len = grams.length();
052        if (n < 1 || n > 3 || len < n) {
053            return null;
054        }
055        if (n == 1) {
056            char ch = grams.charAt(len - 1);
057            if (ch == ' ') {
058                return null;
059            }
060            return Character.toString(ch);
061        } else {
062            return grams.substring(len - n, len);
063        }
064    }
065
066    public static char normalize(char ch) {
067        Character.UnicodeBlock block = Character.UnicodeBlock.of(ch);
068        if (block == UnicodeBlock.BASIC_LATIN) {
069            if (ch < 'A' || (ch < 'a' && ch > 'Z') || ch > 'z') {
070                ch = ' ';
071            }
072        } else if (block == UnicodeBlock.LATIN_1_SUPPLEMENT) {
073            if (LATIN1_EXCLUDED.indexOf(ch) >= 0) {
074                ch = ' ';
075            }
076        } else if (block == UnicodeBlock.GENERAL_PUNCTUATION) {
077            ch = ' ';
078        } else if (block == UnicodeBlock.ARABIC) {
079            if (ch == '\u06cc') {
080                ch = '\u064a';
081            }
082        } else if (block == UnicodeBlock.LATIN_EXTENDED_ADDITIONAL) {
083            if (ch >= '\u1ea0') {
084                ch = '\u1ec3';
085            }
086        } else if (block == UnicodeBlock.HIRAGANA) {
087            ch = '\u3042';
088        } else if (block == UnicodeBlock.KATAKANA) {
089            ch = '\u30a2';
090        } else if (block == UnicodeBlock.BOPOMOFO || block == UnicodeBlock.BOPOMOFO_EXTENDED) {
091            ch = '\u3105';
092        } else if (block == UnicodeBlock.CJK_UNIFIED_IDEOGRAPHS) {
093            if (cjk_map.containsKey(ch)) {
094                ch = cjk_map.get(ch);
095            }
096        } else if (block == UnicodeBlock.HANGUL_SYLLABLES) {
097            ch = '\uac00';
098        }
099        return ch;
100    }
