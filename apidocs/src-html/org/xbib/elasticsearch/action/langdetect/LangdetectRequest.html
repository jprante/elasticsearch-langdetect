002package org.xbib.elasticsearch.action.langdetect;
003
004import java.io.IOException;
005import org.elasticsearch.action.support.single.custom.SingleCustomOperationRequest;
006import org.elasticsearch.common.bytes.BytesReference;
007import org.elasticsearch.common.io.stream.StreamInput;
008import org.elasticsearch.common.io.stream.StreamOutput;
009
010public class LangdetectRequest extends SingleCustomOperationRequest<LangdetectRequest> {
011
012    BytesReference text;
013
014    public LangdetectRequest() {
015    }
016
017    public LangdetectRequest setText(BytesReference text) {
018        this.text = text;
019        return this;
020    }
021    
022    public BytesReference getText() {
023        return text;
024    }
025    
026    @Override
027    public void readFrom(StreamInput in) throws IOException {
028        super.readFrom(in);
029        text = in.readBytesReference();
030    }
031
032    @Override
033    public void writeTo(StreamOutput out) throws IOException {
034        super.writeTo(out);
035        out.writeBytesReference(text);
036    }
037}
